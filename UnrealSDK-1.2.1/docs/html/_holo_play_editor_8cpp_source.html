57 void FHoloPlayEditorModule::AddEditorSettings()
58 {

60 
61  if (SettingsModule)
62  {
63  ISettingsSectionPtr SettingsSection = SettingsModule->RegisterSettings("Project", "Plugins", "HoloPlay",
64  LOCTEXT("HoloPlaySettingsName", "HoloPlay Plugin"),
65  LOCTEXT("HoloPlaySettingsDescription", "Configure the HoloPlay plug-in."),
66  GetMutableDefault<UHoloPlaySettings>()
67  );
68 
69  if (SettingsSection.IsValid())
70  {
71  SettingsSection->OnModified().BindRaw(this, &FHoloPlayEditorModule::OnSettingsSaved);
72  }
73  }
74 }
