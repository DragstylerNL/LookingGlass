72 void FHoloPlayRuntimeModule::StartPlayer(EHoloPlayModeType HoloPlayModeType)
73 {
74  DISPLAY_HOLOPLAY_FUNC_TRACE(HoloPlayLogPlayer);
75 
76  if (bIsPlaying == false)
77  {
78  // Set Current HoloPlayModeType
79  CurrentHoloPlayModeType = HoloPlayModeType;
80  auto HoloPlaySettings = GetDefault<UHoloPlaySettings>();
81 
82  // Start all managers
83  {
84  bool Result = true;
85  auto it = Managers.CreateIterator();
86  while(Result && it)
87  {
88  Result = Result && (*it)->OnStartPlayer(CurrentHoloPlayModeType);
89  ++it;
90  }
91 
92  if (!Result)
93  {
94  UE_LOG(HoloPlayLogPlayer, Verbose, TEXT("Error during StartPlayer managers"));
95  }
96  }
97 
98  switch (HoloPlayModeType)
99  {
100  case EHoloPlayModeType::PlayMode_InSeparateWindow:
101  {
102  bIsDestroyWindowRequested = false;
103 
104  const bool bUseBorderlessWindow = HoloPlaySettings->HoloPlayWindowSettings.bUseBorderlessWindow;
105  EHoloPlayWindowType WindowType = HoloPlaySettings->HoloPlayWindowSettings.WindowType;
106  EAutoCenter AutoCenter = static_cast<EAutoCenter>(HoloPlaySettings->HoloPlayWindowSettings.WindowAutoCenter);
107 
108  static FWindowStyle BorderlessStyle = FWindowStyle::GetDefault();
109  BorderlessStyle
110  .SetActiveTitleBrush(FSlateNoResource())
111  .SetInactiveTitleBrush(FSlateNoResource())
112  .SetFlashTitleBrush(FSlateNoResource())
113  .SetOutlineBrush(FSlateNoResource())
114  .SetBorderBrush(FSlateNoResource())
115  .SetBackgroundBrush(FSlateNoResource())
116  .SetChildBackgroundBrush(FSlateNoResource());
117 
118  FVector2D ClientSize;
119  FVector2D ScreenPosition;
120  auto HoloPlayDisplayMetrics = GetHoloPlayDisplayManager()->GetHoloPlayDisplayMetrics();
121  auto DisplaySettings = GetHoloPlayDisplayManager()->GetDisplaySettings();
122  auto CalibrationSettings = GetHoloPlayDisplayManager()->GetCalibrationSettings();
123  if (HoloPlaySettings->HoloPlayWindowSettings.bAutoPlacementInHoloPlay)
124  {
125  ClientSize.X = CalibrationSettings.ScreenWidth;
126  ClientSize.Y = CalibrationSettings.ScreenHeight;
127 
128  ScreenPosition.X = DisplaySettings.LKGxpos;
129  ScreenPosition.Y = DisplaySettings.LKGypos;
130 
131  AutoCenter = EAutoCenter::None;
132  }
133  else
134  {
135  ClientSize = HoloPlaySettings->HoloPlayWindowSettings.ClientSize;
136  ScreenPosition = HoloPlaySettings->HoloPlayWindowSettings.ScreenPosition;
137  }
138 
139  HoloPlayWindow = SNew(SWindow)
140  .Type(EWindowType::GameWindow)
141  .Style(bUseBorderlessWindow ? &BorderlessStyle : &FCoreStyle::Get().GetWidgetStyle<FWindowStyle>("Window"))
142  .ClientSize(ClientSize)
143  .AdjustInitialSizeAndPositionForDPIScale(false)
144  .Title(FText::FromString("HoloPlay window"))
145  .FocusWhenFirstShown(true)
146  .ScreenPosition(ScreenPosition)
147  .UseOSWindowBorder(HoloPlaySettings->HoloPlayWindowSettings.bUseOSWindowBorder)
148  .CreateTitleBar(!bUseBorderlessWindow)
149  .LayoutBorder(bUseBorderlessWindow ? FMargin(0) : FMargin(5, 5, 5, 5))
150  .AutoCenter(AutoCenter)
151  .SaneWindowPlacement(AutoCenter == EAutoCenter::None)
152  .SizingRule(ESizingRule::UserSized);
153 
154  // 1. Add new window
155  const bool bShowImmediately = false;
156  FSlateApplication::Get().AddWindow(HoloPlayWindow.ToSharedRef(), bShowImmediately);
157 
158  // 2. Set window mode
159  EWindowMode::Type WinMode = static_cast<EWindowMode::Type>(WindowType);
160  // Do not set fullscreen mode here
161  // The window mode will be set properly later
162  if (WinMode == EWindowMode::Fullscreen)
163  {
164  HoloPlayWindow->SetWindowMode(EWindowMode::WindowedFullscreen);
165  }
166  else
167  {
168  HoloPlayWindow->SetWindowMode(WinMode);
169  }
170 
171  // 3. Show window
172  // No need to show window in off-screen rendering mode as it does not render to screen
173  if (FSlateApplication::Get().IsRenderingOffScreen())
174  {
175  FSlateApplicationBase::Get().GetRenderer()->CreateViewport(HoloPlayWindow.ToSharedRef());
176  }
177  else
178  {
179  HoloPlayWindow->ShowWindow();
180  }
181 
182  // 4. Tick now to force a redraw of the window and ensure correct fullscreen application
183  FSlateApplication::Get().Tick();
184 
185  // 5. Add viewport to window
186  // Create and assign viewport for Window. It could be with rendering directly to separate window or not if UE4 window, not native OS
187  bool RenderDirectlyToWindowInSeparateWindow = false;
188  if (!HoloPlaySettings->HoloPlayWindowSettings.bUseOSWindowBorder && !bUseBorderlessWindow)
189  {
190  RenderDirectlyToWindowInSeparateWindow = false;
191  }
192  else
193  {
194  RenderDirectlyToWindowInSeparateWindow = HoloPlaySettings->HoloPlayWindowSettings.RenderDirectlyToWindowInSeparateWindow;
195  }
196  HoloPlayViewport = SNew(SHoloPlayViewport).RenderDirectlyToWindow(RenderDirectlyToWindowInSeparateWindow);
197  HoloPlayViewport->GetHoloPlayViewportClient()->SetViewportWindow(HoloPlayWindow);
198  HoloPlayWindow->SetContent(HoloPlayViewport.ToSharedRef());
199  HoloPlayWindow->SlatePrepass();
200  if (WinMode != HoloPlayWindow->GetWindowMode())
201  {
202  HoloPlayWindow->SetWindowMode(WinMode);
203  HoloPlayWindow->ReshapeWindow(ScreenPosition, ClientSize);
204  FVector2D NewViewportSize = HoloPlayWindow->GetViewportSize();
205  HoloPlayViewport->GetSceneViewport()->UpdateViewportRHI(false, NewViewportSize.X, NewViewportSize.Y, WinMode, PF_Unknown);
206  HoloPlayViewport->GetSceneViewport()->Invalidate();
207 
208  // Resize backbuffer
209  FVector2D NewBackbufferSize = HoloPlayWindow->IsMirrorWindow() ? ClientSize : NewViewportSize;
210  FSlateApplicationBase::Get().GetRenderer()->UpdateFullscreenState(HoloPlayWindow.ToSharedRef(), NewBackbufferSize.X, NewBackbufferSize.Y);
211  }
212 
213  // 6. Add window delegates
214  OnWindowClosedDelegate.BindRaw(this, &FHoloPlayRuntimeModule::OnWindowClosed);
215  HoloPlayWindow->SetOnWindowClosed(OnWindowClosedDelegate);
216 
217  break;
218  }
219  case EHoloPlayModeType::PlayMode_InMainViewport:
220  {
221  if (GEngine->GameViewport)
222  {
223  bool bRenderDirectlyToWindow = (GIsEditor) ? false : true;
224 
225  TSharedPtr<SWindow> GameViewportWindow = GEngine->GameViewport->GetWindow();
226  FString WindowTitle = GameViewportWindow->GetTitle().ToString();
227  UMovieSceneCapture* MovieSceneCapture = HoloPlay::GetMovieSceneCapture();
228 
229  FVector2D ClientSize;
230  FVector2D ScreenPosition;
231  FMonitorInfo FirstHoloPlayDisplayInfo;
232  auto HoloPlayDisplayMetrics = GetHoloPlayDisplayManager()->GetHoloPlayDisplayMetrics();
233  auto DisplaySettings = GetHoloPlayDisplayManager()->GetDisplaySettings();
234  auto CalibrationSettings = GetHoloPlayDisplayManager()->GetCalibrationSettings();
235  if (HoloPlaySettings->HoloPlayWindowSettings.bAutoPlacementInHoloPlay)
236  {
237  ClientSize.X = CalibrationSettings.ScreenWidth;
238  ClientSize.Y = CalibrationSettings.ScreenHeight;
239 
240  ScreenPosition.X = DisplaySettings.LKGxpos;
241  ScreenPosition.Y = DisplaySettings.LKGypos;
242 
243  bool bRequesWindowAutoPlacement = false;
244 #if WITH_EDITOR
245  auto LastExecutedPlayModeType = GetDefault<ULevelEditorPlaySettings>()->LastExecutedPlayModeType;
246 
247  // If Movie capturing
248  if ( (MovieSceneCapture != nullptr && MovieSceneCapture->IsRooted()) || bIsCaptureStandaloneMovie)
249  {
250  bRenderDirectlyToWindow = false;
251  }
252  // if Editor Floating window
253  else if (LastExecutedPlayModeType == EPlayModeType::PlayMode_InEditorFloating)
254  {
255  FMargin Border = GameViewportWindow->GetWindowBorderSize();
256  ScreenPosition.X -= Border.Left;
257  ClientSize.X += Border.Left + Border.Right;
258  ScreenPosition.Y -= Border.Top;
259  ClientSize.Y += Border.Top + Border.Bottom;
260 
261  GameViewportWindow->ReshapeWindow(ScreenPosition, ClientSize);
262  }
263  // Check placement if bIsStandaloneGame or bIsGameMode
264  else if (bIsStandaloneGame || bIsGameMode)
265  {
266 
267  FSystemResolution::RequestResolutionChange(ClientSize.X, ClientSize.Y, GameViewportWindow->GetWindowMode());
268  GameViewportWindow->ReshapeWindow(ScreenPosition, ClientSize);
269  }
270 #else
271  {
272  FSystemResolution::RequestResolutionChange(ClientSize.X, ClientSize.Y, GameViewportWindow->GetWindowMode());
273  GameViewportWindow->ReshapeWindow(ScreenPosition, ClientSize);
274  }
275 #endif
276  }
277  GEngine->GameViewport->OnCloseRequested().AddRaw(this, &FHoloPlayRuntimeModule::OnGameViewportCloseRequested);
278 
279  // Create and assign viewport to window
280  HoloPlayViewport = SNew(SHoloPlayViewport).RenderDirectlyToWindow(bRenderDirectlyToWindow);
281  HoloPlayViewport->GetHoloPlayViewportClient()->SetViewportWindow(GEngine->GameViewport->GetWindow());
282 
283  GEngine->GameViewport->bDisableWorldRendering = true;
284  GEngine->GameViewport->AddViewportWidgetContent(
285  HoloPlayViewport.ToSharedRef()
286  );
287  }
288  break;
289  }
290  default:
291  return;
292  }
293 
294  bIsPlaying = true;
295  }

298 void FHoloPlayRuntimeModule::StopPlayer()
299 {
300  DISPLAY_HOLOPLAY_FUNC_TRACE(HoloPlayLogPlayer);
301 
302  if (bIsPlaying == true)
303  {
304  // Stop all managers
305  for (auto Mng : Managers)
306  {
307  Mng->OnStopPlayer();
308  }
309 
310  auto PlayModeType = GetDefault<UHoloPlaySettings>()->HoloPlayWindowSettings.LastExecutedPlayModeType;
311 
312  // Stop PlayInStandalon first
313  if (bIsStandaloneGame || bIsCaptureStandaloneMovie || GetDefault<UHoloPlaySettings>()->HoloPlayWindowSettings.bLockInMainViewport)
314  {
315  PlayModeType = EHoloPlayModeType::PlayMode_InMainViewport;
316  }
317 
318  switch (PlayModeType)
319  {
320  case EHoloPlayModeType::PlayMode_InSeparateWindow:
321  {
322  if (bIsDestroyWindowRequested == false)
323  {
324  bIsDestroyWindowRequested = true;
325 
326  HoloPlayWindow->RequestDestroyWindow();
327 
328  // Stop player
329  OnWindowClosedDelegate.Unbind();
330 
331  HoloPlayViewport.Reset();
332  HoloPlayWindow.Reset();
333  }
334 
335  break;
336  }
337  case EHoloPlayModeType::PlayMode_InMainViewport:
338  {
339  if (GEngine->GameViewport)
340  {
341  GEngine->GameViewport->bDisableWorldRendering = false;
342  GEngine->GameViewport->RemoveViewportWidgetContent(
343  HoloPlayViewport.ToSharedRef()
344  );
345 
346  HoloPlayViewport.Reset();
347  }
348 
349  break;
350  }
351  default:
352  return;
353  }
354 
355  bIsPlaying = false;
356  }

385 void FHoloPlayRuntimeModule::StartPlayerSeparateProccess()
386 {
387  auto HoloPlaySettings = GetDefault<UHoloPlaySettings>();
388  auto LastExecutedPlayModeType = (HoloPlaySettings->HoloPlayWindowSettings.bLockInMainViewport) ?
389  EHoloPlayModeType::PlayMode_InMainViewport : HoloPlaySettings->HoloPlayWindowSettings.LastExecutedPlayModeType;
390 
391 #if WITH_EDITOR
392  // Try to parse if this is standalone mode
393  {
394  FString StandaloneSessionName = "Play in Standalone Game";
395  bIsStandaloneGame = FApp::GetSessionName() == StandaloneSessionName;
396  if (bIsStandaloneGame)
397  {
398  StartPlayer(EHoloPlayModeType::PlayMode_InMainViewport);
399  return;
400  }
401  }
402 
403  // Try to parse separate process in capture scene
404  {
405  FString CaptureStandaloneMovie;
406  // Try to read screen model from command line
407  FParse::Value(FCommandLine::Get(), TEXT("-MovieSceneCaptureManifest="), CaptureStandaloneMovie);
408 
409  bIsCaptureStandaloneMovie = !CaptureStandaloneMovie.IsEmpty();
410  if (bIsCaptureStandaloneMovie)
411  {
412  StartPlayer(EHoloPlayModeType::PlayMode_InMainViewport);
413  return;
414  }
415  }
416 
417  // Play in editor game mode
418  {
419 
420  bool OnOff = true;
421  bIsGameMode = FParse::Bool(FCommandLine::Get(), TEXT("-game"), OnOff);
422  if (bIsGameMode)
423  {
424  StartPlayer(LastExecutedPlayModeType);
425  return;
426  }
427  }
428 
429 #else
430  StartPlayer(LastExecutedPlayModeType);
431 #endif
432 }
